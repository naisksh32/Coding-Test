# 빙산
##### Tag: 구현, 그래프 이론, 그래프 탐색, 너비 우선 탐색, 깊이 우선 탐색
##### 풀이 일자: 2026년 2월 16일

## 문제 탐색하기
빙산이 2개 이상으로 나뉘는 그 순간을 찾기

### 해결요소
1. 빙산은 인접한 0에 따라서 녹는 비율이 다름 -> 따로 임시 배열을 만들었다가 계산 후 적용
2. 모든 빙산이 1덩어리에서 녹을 때에는 0을 반환

## 코드 설계하기
```
// 빙산의 주변을 녹이는 구현 함수
void turn() {
	int tmp_board[301][301] = { 0 };

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {

			if (board[i][j] != 0) {
				int zero_area = 0;

				for (int d = 0; d < 4; d++) {
					int ny = i + dy[d];
					int nx = j + dx[d];

					if (ny >= N || ny < 0 || nx >= M || nx < 0) continue;

					if (board[ny][nx] == 0) zero_area++;
				}

				tmp_board[i][j] = board[i][j] - zero_area;
				if (tmp_board[i][j] < 0) tmp_board[i][j] = 0;
				
			}

		}
	}

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			board[i][j] = tmp_board[i][j];
		}
	}

}

// 빙산의 덩어리를 확인하는 BFS
int bfs() {
	memset(visited, 0, sizeof(visited));
	int conti = 0;

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {

			if (board[i][j] != 0 && visited[i][j] == 0) {
				conti++;

				queue<Pos> q;
				q.push({ i, j });
				visited[i][j] = conti;

				while (!q.empty()) {
					Pos now = q.front();
					q.pop();

					for (int d = 0; d < 4; d++) {
						int ny = now.y + dy[d];
						int nx = now.x + dx[d];

						if (ny >= N || ny < 0 || nx >= M || nx < 0) continue;
						if (board[ny][nx] == 0) continue;
						if (visited[ny][nx] != 0) continue;

						visited[ny][nx] = conti;
						q.push({ ny, nx });

					}
				}

			}

		}
	}

	return conti;
}
```
### 시간 복잡도
O(V^2)  
인접 행렬로 관리  
V는 최대 300 이므로 90,000 < 1억 임으로 연산 가능

int N, M;
int board[301][301];
int visited[301][301];

struct Pos {
	int y, x;
};

int dy[] = { -1, 0, 1, 0 };
int dx[] = { 0, 1, 0, -1 };

void turn() {
	int tmp_board[301][301] = { 0 };

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {

				}

				tmp_board[i][j] = board[i][j] - zero_area;
				if (tmp_board[i][j] < 0) tmp_board[i][j] = 0;
				
			}

		}
	}

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			board[i][j] = tmp_board[i][j];
		}
	}

}

int bfs() {
	memset(visited, 0, sizeof(visited));
	int conti = 0;

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {

			if (board[i][j] != 0 && visited[i][j] == 0) {

					q.pop();

						visited[ny][nx] = conti;

	}

void print_board() {
